import os,subprocess,atexit
from jsonutils import jsonutils
os.chdir(os.path.dirname(__file__))

models = [model for model in os.listdir(os.path.join(os.path.dirname(__file__),'models'))]
models.sort(key=lambda x: str(os.path.basename(x)).split('.')[:2],reverse=True)
model = 'latest'

ai = Ai(datafp=os.path.join(os.path.dirname(__file__),'models',models[0] if model == 'latest' else model),device='cpu')
#tts = TTS(device=None,lang='it',tld='com',slow=False)
listener = Listener(device=None,lang='it-IT')

def remove():
    path = os.environ['PATH'].split(';')
    tools = os.path.join(os.path.dirname(__file__),'tools')
    if tools in path: path.remove(tools)
    path = ';'.join(path)
    subprocess.Popen(['setx', 'PATH', path], shell=True)

#remove the directory from system variables at exit.
atexit.register(remove)
#add the directory from system variables at exit.
subprocess.Popen(['setx', 'PATH', f"{os.environ['PATH']};{os.path.join(os.path.dirname(__file__),'tools')}"],stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE, shell=True)
